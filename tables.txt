CREATE TABLE applicant(
	id int(13) not null AUTO_INCREMENT PRIMARY KEY,
    first_name varCHAR(255) not null,
    last_name varCHAR(255) not null,
    other_names varCHAR(255),
    phone_number varCHAR(255) not null,
    member_code varCHAR(255),
    staff_id varCHAR(255),
    work_place varCHAR(255) not null,
    member_status varCHAR(255),
    purpose varCHAR(255) not null,
    loan_amount int not null,
    loan_amount_interest int not null,
    apply_date date DEFAULT CURRENT_DATE,
    mode_of_payment varCHAR(255) not null,
    guarantor_fullname_first varCHAR(255) not null,
    guarantor_phone_first varCHAR(255) not null,
    guarantor_staffnum_first varCHAR(255) not null,
    guaranteed_amount_first varCHAR(255) not null,
    guarantor_fullname_second varCHAR(255) not null,
    guarantor_phone_second varCHAR(255) not null,
    guarantor_staffnum_second varCHAR(255) not null,
    guaranteed_amount_second varCHAR(255) not null,
    guarantor_fullname_thrid varCHAR(255) not null,
    guarantor_phone_third varCHAR(255) not null,
    guarantor_staffnum_third varCHAR(255) not null,
    guaranteed_amount_third varCHAR(255) not null,
    guarantor_fullname_fourth varCHAR(255) not null,
    guarantor_phone_fourth varCHAR(255) not null,
    guarantor_staffnum_fourth varCHAR(255) not null,
    guaranteed_amount_fourth varCHAR(255) not null,
    recipient_name varCHAR(255) not null,
    first_due_date date,
    first_due_date_status varchar(255),
    first_due_recipient varchar(255),
    second_due_date date,
    second_due_date_status varchar(255),
    second_due_recipient varchar(255),
    third_due_date date,
    third_due_date_status varchar(255),
    third_due_recipient varchar(255),
    fourth_due_date date,
    fourth_due_date_status varchar(255),
    fourth_due_recipient varchar(255),
    fifth_due_date date,
    fifth_due_date_status varchar(255),
    fifth_due_recipient varchar(255),
    sixth_due_date date,
    sixth_due_date_status varchar(255),
    sixth_due_recipient varchar(255),
    approved_by varCHAR(255),
    approval_status varCHAR(255) DEFAULT "pending",
    loan_status varCHAR(255),
    day_approved datetime,
    day_granted datetime,
    granted_by varCHAR(255)
);
